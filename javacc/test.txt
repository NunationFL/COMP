import io;

class Fac
{
    public int ComputeFac(int num)
    {
        int num_aux;
        if (num < 1) num_aux = 1;
        else num_aux = num * (this.ComputeFac(num-1));
        return num_aux;
    }

    public static void main(String[] args)
    {
        int i;
        i = 0;
        io.println(new Fac().ComputeFac(10)); 

        while (i < 3)
        {
            i = 3;
        }

    }
}
